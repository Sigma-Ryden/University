#include <iostream>
#include <cmath>

double function(double x)
{
    double fx{};

    return fx;
}

double integration()
{
    const double N{ pow(10, 6) };
    double sum{};

    for (unsigned int i = 1; i < N; i++)
    {
        sum += function(i / N);
    }

    return sum/N;
}

int main()
{
    using std::cout;
    using std::endl;
   
    cout << "Result: " << integration() << endl;

    return 0;
}
